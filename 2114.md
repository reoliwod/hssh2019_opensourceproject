import random 

def splicing(seq) : #스플라이싱 과정을 나타내는 함수
    base = input('Splicing을 원하시는 Sequence를 입력해주세요')
    for i in range(1,len(seq)-len(base)+1) :
        if seq[i:i+len(base)+1]== base :
            seq == seq[1:i] + seq[i+1]
    return seq
